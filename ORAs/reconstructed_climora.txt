  const [data, setData] = useState(null);
  useEffect(() => { let mounted = true; getWeatherData(loc.lat, loc.lon).then(d => { if (mounted) setData(d); }); return () => { mounted = false; }; }, [loc]);

  if (!data) return <div className="h-24 glass-panel rounded-[3rem] animate-pulse mb-3 bg-white/5" />;

  const Icon = getSavedLocationIcon(data.current.weather_code, data.current.is_day);
  const liveStatus = getDetailedClimateLabel(data.current.weather_code, data.current.is_day);

  return (
    <div onClick={() => onClick(loc)} className="glass-panel rounded-[3rem] p-6 mb-4 relative overflow-hidden group active:scale-[0.98] transition-all cursor-pointer hover:bg-white/5 shadow-xl">
      <div className="absolute right-[-20px] top-[-20px] opacity-10 group-hover:scale-110 transition-transform"><Icon size={120} className="icon-glow" /></div>
      <div className="relative z-10 flex justify-between items-center w-full">
        <div className="flex flex-col">
          <div className="text-sm font-bold">{loc.name} <span className="text-white/50 text-xs">{loc.country}</span></div>
          <div className="text-xs text-white/50 mt-1">{liveStatus}</div>
        </div>
        <div className="flex items-center gap-4">
          <div className="text-3xl font-light">{Math.round(data.current.temperature_2m)}°</div>
          <button onClick={(e) => { e.stopPropagation(); onDelete(loc.id); }} className="p-3 bg-red-500/10 text-red-400 rounded-full hover:bg-red-500 hover:text-white transition-colors z-20 shadow-inner">

const CelestialArc = ({ sunrise, sunset }) => {
    const now = new Date();
    const sr = new Date(sunrise);
    const ss = new Date(sunset);
    
    const dayDuration = ss - sr;
    const timePassed = now - sr;
    
    let progress = timePassed / dayDuration;
    if (progress < 0) progress = 0;
    if (progress > 1) progress = 1;
    
    const leftPos = progress * 100;
    const bottomPos = Math.sin(progress * Math.PI) * 100;
    
    const isDay = now >= sr && now <= ss;
    
    return (
        <div className="relative h-48 w-full flex items-end justify-center mb-8">
            <div className="absolute w-full h-[300px] border-t-2 border-dashed border-white/10 rounded-full top-0" />
            <div className="w-full flex justify-between px-8 md:px-20 pb-2 relative z-10">
               <div className="flex flex-col items-center gap-2"><Sunrise size={32} className="text-orange-300 icon-glow" /><div className="text-center"><span className="text-xl font-bold block">{formatTime(sunrise)}</span><span className="text-[10px] uppercase opacity-50 tracking-widest">Sunrise</span></div></div>
               <div className="flex flex-col items-center gap-2"><Sunset size={32} className="text-purple-400 icon-glow" /><div className="text-center"><span className="text-xl font-bold block">{formatTime(sunset)}</span><span className="text-[10px] uppercase opacity-50 tracking-widest">Sunset</span></div></div>
            </div>
            
            {isDay ? (
                <div 
                    className="absolute w-12 h-12 bg-yellow-400 rounded-full blur-[15px] shadow-[0_0_60px_#facc15] transition-all duration-1000 ease-linear"
                    style={{ left: `calc(${leftPos}% - 24px)`, bottom: `calc(${bottomPos}% - 24px)` }}
                />
            ) : (
                <div 
                    className="absolute w-10 h-10 bg-slate-200 rounded-full blur-[10px] shadow-[0_0_40px_#fff] transition-all duration-1000 ease-linear"
                    style={{ left: '50%', bottom: '10%' }}
                />
            )}
        </div>
    );
};
